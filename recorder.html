const state = {

  mediaStream: null,

  mediaRecorder: null,

  isRecording: false,

  isStopping: false,

  sessionId: '',

  meetingTitle: '',

  startedAt: null,

  startedAtText: '',

  chunkIndex: 0,

  uploadQueue: [],

  uploading: false,

  uploadedCount: 0,

  failedCount: 0,

  timerInterval: null,

  currentMimeType: '',

  finalizeDone: false

};


const el = {

  meetingTitle: document.getElementById('meetingTitle'),

  segmentSeconds: document.getElementById('segmentSeconds'),

  audioBits: document.getElementById('audioBits'),

  healthBtn: document.getElementById('healthBtn'),

  resetBtn: document.getElementById('resetBtn'),

  stopBtn: document.getElementById('stopBtn'),

el.stopBtn.addEventListener('click', stopRecordingAndFinalize);


window.addEventListener('beforeunload', function(e) {

  if (state.isRecording || state.uploading || state.uploadQueue.length > 0) {

    e.preventDefault();

    e.returnValue = '';

  }

});


function validateGasUrl() {

  if (!GAS_WEB_APP_URL || GAS_WEB_APP_URL.indexOf('https://') !== 0 || GAS_WEB_APP_URL.indexOf('ここに') !== -1) {

    throw new Error('GAS_WEB_APP_URL が未設定です。GASのWebアプリURLを recorder.html に貼り付けてください。');

  }

}


async function healthCheck() {

  try {

    validateGasUrl();

    setStatus('GAS接続テスト中...', 'warn');


    const res = await postToGas({

      action: 'health'

    });


    if (!res.ok) {

      throw new Error(res.error || 'GAS接続テストに失敗しました。');

    }


    setStatus(

      'GAS接続OK\n' +

      'フォルダ: ' + res.folderName + '\n' +

      'モデル: ' + res.model,

      'ok'

    );


    addLog('GAS接続テスト成功: ' + JSON.stringify(res));


  } catch (err) {

    setStatus('GAS接続エラー: ' + err.message, 'error');

    addLog('GAS接続エラー: ' + err.stack);

  }

}


async function startRecording() {

  try {

    validateGasUrl();


    const title = el.meetingTitle.value.trim();

    if (!title) {

      alert('議事録名を入力してください。');

      return;

    }


    if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {

      throw new Error('このブラウザは getUserMedia に対応していません。Chrome最新版で開いてください。');

    }


    state.sessionId = createSessionId();

    state.meetingTitle = title;

    state.startedAt = new Date();

    state.startedAtText = formatDateTimeForFile(state.startedAt);

    state.chunkIndex = 0;

    state.uploadQueue = [];

    state.uploadedCount = 0;

    state.failedCount = 0;

    state.isRecording = true;

    state.isStopping = false;

    state.finalizeDone = false;


    el.resultArea.innerHTML = '';

    el.startBtn.disabled = true;

    el.stopBtn.disabled = false;

    el.meetingTitle.disabled = true;

    el.segmentSeconds.disabled = true;

    el.audioBits.disabled = true;


    setStatus('マイク許可を確認中...', 'warn');


    state.mediaStream = await navigator.mediaDevices.getUserMedia({

      audio: {

        echoCancellation: true,

        noiseSuppression: true,

        autoGainControl: true,

        channelCount: 1

      },

      video: false

    });


    state.currentMimeType = chooseSupportedMimeType();


    if (!state.currentMimeType) {

      throw new Error('このブラウザで利用可能な録音形式が見つかりません。');

    }


    addLog('録音形式: ' + state.currentMimeType);

    setStatus('録音中...', 'ok');


    startTimer();

    startNextSegment();


  } catch (err) {

    await cleanupRecording();

    setStatus('録音開始に失敗しました: ' + err.message, 'error');

    addLog('録音開始エラー: ' + err.stack);

    restoreControls();

  }

}


function startNextSegment() {

  if (!state.isRecording || state.isStopping) return;


  const bits = Number(el.audioBits.value || 48000);

  const segmentMs = Number(el.segmentSeconds.value || 60) * 1000;


  const chunks = [];


  let options = {

    mimeType: state.currentMimeType,

    audioBitsPerSecond: bits

  };


  try {

    state.mediaRecorder = new MediaRecorder(state.mediaStream, options);

  } catch (err) {

    addLog('指定mimeTypeでMediaRecorder作成失敗。mimeType指定なしで再試行: ' + err.message);

    state.mediaRecorder = new MediaRecorder(state.mediaStream, {

      audioBitsPerSecond: bits

    });

    state.currentMimeType = state.mediaRecorder.mimeType || state.currentMimeType;

  }


  const thisChunkIndex = state.chunkIndex++;


  state.mediaRecorder.ondataavailable = function(event) {

    if (event.data && event.data.size > 0) {

      chunks.push(event.data);

    }

  };


  state.mediaRecorder.onerror = function(event) {

    addLog('MediaRecorderエラー: ' + JSON.stringify(event));

  };


  state.mediaRecorder.onstop = async function() {

    try {

      const blob = new Blob(chunks, {

        type: state.mediaRecorder.mimeType || state.currentMimeType

      });


      if (blob.size > 0) {

        enqueueUpload({

          chunkIndex: thisChunkIndex,

          blob: blob,

          mimeType: blob.type || state.currentMimeType

        });

      } else {

        addLog('空の録音チャンクをスキップ: ' + thisChunkIndex);

      }


    } catch (err) {

      state.failedCount++;

      addLog('チャンク生成エラー: ' + err.stack);

    }


    if (state.isRecording && !state.isStopping) {

      startNextSegment();

    }

  };


  state.mediaRecorder.start();


  addLog('チャンク録音開始: ' + thisChunkIndex);


  setTimeout(function() {

    if (

      state.mediaRecorder &&

      state.mediaRecorder.state === 'recording' &&

      state.isRecording &&

      !state.isStopping

    ) {

      state.mediaRecorder.stop();

    }

  }, segmentMs);

}


async function stopRecordingAndFinalize() {

  try {

    state.isStopping = true;

    state.isRecording = false;


    el.stopBtn.disabled = true;

    setStatus('録音停止中。残りデータをアップロードしています...', 'warn');


    if (state.mediaRecorder && state.mediaRecorder.state === 'recording') {

      state.mediaRecorder.stop();

    }


    if (state.mediaStream) {

      state.mediaStream.getTracks().forEach(function(track) {

        track.stop();

      });

    }


    stopTimer();


    await waitForUploadsToFinish();


    setStatus('議事録を作成中...', 'warn');


    const finalizeRes = await postToGas({

      action: 'finalize',

      sessionId: state.sessionId,

      meetingTitle: state.meetingTitle,

      startedAtText: state.startedAtText,

      clientInfo: getClientInfo()

    });


    if (!finalizeRes.ok) {

      throw new Error(finalizeRes.error || '議事録作成に失敗しました。');

    }


    state.finalizeDone = true;


    setStatus(

      '完了\n' +

      '音声チャンク: ' + finalizeRes.audioChunks + '\n' +

      '文字起こしチャンク: ' + finalizeRes.transcriptChunks + '\n' +

      'エラー数: ' + finalizeRes.errorFiles,

      finalizeRes.errorFiles > 0 ? 'warn' : 'ok'

    );


    showResult(finalizeRes.documentUrl);

    addLog('議事録作成完了: ' + JSON.stringify(finalizeRes));


  } catch (err) {

    setStatus('停止後処理エラー: ' + err.message, 'error');

    addLog('停止後処理エラー: ' + err.stack);


  } finally {

    await cleanupRecording();

    restoreControls();

  }

}


function enqueueUpload(item) {

  state.uploadQueue.push(item);

  addLog('アップロード待ち追加: chunk=' + item.chunkIndex + ', size=' + item.blob.size);

  processUploadQueue();

}


async function processUploadQueue() {

  if (state.uploading) return;

  state.uploading = true;


  try {

    while (state.uploadQueue.length > 0) {

      const item = state.uploadQueue.shift();


      try {

        await uploadChunkWithRetry(item, 3);

        state.uploadedCount++;

      } catch (err) {

        state.failedCount++;

        addLog('アップロード失敗 chunk=' + item.chunkIndex + ': ' + err.stack);

      }


      updateProgress();

    }

  } finally {

    state.uploading = false;

  }

}


async function uploadChunkWithRetry(item, maxRetry) {

  let lastError = null;


  for (let attempt = 1; attempt <= maxRetry; attempt++) {

    try {

      setStatus(

        'アップロード中...\n' +

        'チャンク: ' + item.chunkIndex + '\n' +

        '試行: ' + attempt + '/' + maxRetry,

        'warn'

      );


      const audioBase64 = await blobToBase64(item.blob);


      const res = await postToGas({

        action: 'uploadChunk',

        sessionId: state.sessionId,

        meetingTitle: state.meetingTitle,

        startedAtText: state.startedAtText,

        chunkIndex: item.chunkIndex,

        mimeType: item.mimeType,

        audioBase64: audioBase64,

        clientInfo: getClientInfo()

      });


      if (!res.ok) {

        throw new Error(res.error || 'GAS側でアップロード処理に失敗しました。');

      }


      if (!res.geminiOk) {

        addLog(

          '音声保存成功・Gemini文字起こし失敗 chunk=' +

          item.chunkIndex +

          ': ' +

          res.geminiError

        );

      } else {

        addLog('アップロード・文字起こし成功 chunk=' + item.chunkIndex);

      }


      return res;


    } catch (err) {

      lastError = err;

      addLog('リトライ対象エラー chunk=' + item.chunkIndex + ', attempt=' + attempt + ': ' + err.message);

      await sleep(1200 * attempt);

    }

  }


  throw lastError;

}


async function waitForUploadsToFinish() {

  while (state.uploading || state.uploadQueue.length > 0) {

    setStatus(

      'アップロード完了待ち...\n' +

      '残り: ' + state.uploadQueue.length + '\n' +

      '完了: ' + state.uploadedCount + '\n' +

      '失敗: ' + state.failedCount,

      'warn'

    );

    await sleep(800);

  }

}


async function postToGas(payload) {

  const res = await fetch(GAS_WEB_APP_URL, {

    method: 'POST',

    headers: {

      'Content-Type': 'text/plain;charset=utf-8'

    },

    body: JSON.stringify(payload),

    redirect: 'follow'

  });


  const text = await res.text();


  try {

    return JSON.parse(text);

  } catch (err) {

    throw new Error('GASレスポンスをJSON解析できません: ' + text.substring(0, 500));

  }

}


function chooseSupportedMimeType() {

  const candidates = [

    'audio/mp4',

    'audio/aac',

    'audio/ogg;codecs=opus',

    'audio/webm;codecs=opus',

    'audio/webm'

  ];


  if (!window.MediaRecorder || !MediaRecorder.isTypeSupported) {

    return '';

  }


  for (const type of candidates) {

    if (MediaRecorder.isTypeSupported(type)) {

      return type;

    }

  }


  return '';

}


function blobToBase64(blob) {

  return new Promise(function(resolve, reject) {

    const reader = new FileReader();


    reader.onloadend = function() {

      const dataUrl = reader.result;

      const base64 = String(dataUrl).split(',')[1];

      resolve(base64);

    };


    reader.onerror = function() {

      reject(new Error('BlobのBase64変換に失敗しました。'));

    };


    reader.readAsDataURL(blob);

  });

}


function startTimer() {

  stopTimer();


  state.timerInterval = setInterval(function() {

    if (!state.startedAt) return;


    const diff = Date.now() - state.startedAt.getTime();

    const totalSec = Math.floor(diff / 1000);

    const h = Math.floor(totalSec / 3600);

    const m = Math.floor((totalSec % 3600) / 60);

    const s = totalSec % 60;


    el.timer.textContent =

      String(h).padStart(2, '0') + ':' +

      String(m).padStart(2, '0') + ':' +

      String(s).padStart(2, '0');

  }, 500);

}


function stopTimer() {

  if (state.timerInterval) {

    clearInterval(state.timerInterval);

    state.timerInterval = null;

  }

}


async function cleanupRecording() {

  try {

    if (state.mediaRecorder && state.mediaRecorder.state === 'recording') {

      state.mediaRecorder.stop();

    }

  } catch (err) {}


  try {

    if (state.mediaStream) {

      state.mediaStream.getTracks().forEach(function(track) {

        track.stop();

      });

    }

  } catch (err) {}


  state.mediaRecorder = null;

  state.mediaStream = null;

  state.isRecording = false;

  state.isStopping = false;

}

}


function resetScreen() {

  if (state.isRecording || state.uploading || state.uploadQueue.length > 0) {

    alert('録音中またはアップロード中はリセットできません。');

    return;

  }


  el.timer.textContent = '00:00:00';

  el.status.className = 'status';

  el.status.textContent = '待機中';

  el.meterBar.style.width = '0%';

  el.resultArea.innerHTML = '';

  el.log.textContent = '';

}

  el.status.textContent = message;

  el.status.className = 'status ' + (type || '');

}


function updateProgress() {

  const total = state.uploadedCount + state.failedCount + state.uploadQueue.length + (state.uploading ? 1 : 0);

  let percent = 0;


  if (state.chunkIndex > 0) {

    percent = Math.min(100, Math.round((state.uploadedCount / state.chunkIndex) * 100));

  }


  el.meterBar.style.width = percent + '%';


  if (state.isRecording) {

    setStatus(

      '録音中...\n' +

      '録音チャンク: ' + state.chunkIndex + '\n' +

      'アップロード完了: ' + state.uploadedCount + '\n' +

      'アップロード待ち: ' + state.uploadQueue.length + '\n' +

      '失敗: ' + state.failedCount,

      'ok'

    );

  }

}

function createSessionId() {

  return 's' + Date.now() + '_' + Math.random().toString(36).substring(2, 10);

}


function formatDateTimeForFile(date) {

  const y = date.getFullYear();

  const mo = String(date.getMonth() + 1).padStart(2, '0');

  const d = String(date.getDate()).padStart(2, '0');

  const h = String(date.getHours()).padStart(2, '0');

  const mi = String(date.getMinutes()).padStart(2, '0');

  const s = String(date.getSeconds()).padStart(2, '0');


  return y + mo + d + '_' + h + mi + s;

}

    pageUrl: location.href

  };

}


function sleep(ms) {

  return new Promise(function(resolve) {

    setTimeout(resolve, ms);

  });

}


function escapeHtml(str) {

  return String(str)

    .replaceAll('&', '&amp;')

    .replaceAll('<', '&lt;')

    .replaceAll('>', '&gt;')

    .replaceAll('"', '&quot;')

    .replaceAll("'", '&#039;');

}
